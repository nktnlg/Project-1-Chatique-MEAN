  [(ngModel)]="wsMsg"
  (change)="sendToServer($event)">

  ts


import {webSocket} from "rxjs/webSocket";

    wsMsg = 'hello'
    subject = webSocket('ws://localhost:8889/')

      sendToServer($event){
        this.subject.subscribe();
        this.subject.next(this.wsMsg);
        this.subject.complete()
      }
